## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| kore-llm → LLM provider APIs | API keys sent over HTTPS; keys must never appear in logs or LLMChunk content |
| LLM provider response → LLMChunk | Provider may return unexpected tool call formats; must sanitize before emitting |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | API key logging | mitigate | API keys received as constructor parameters; never logged in any adapter; KDoc explicitly documents this requirement |
| T-05-02 | Tampering | LLM tool call arguments | accept | Validation is responsibility of ToolProvider implementations; adapters pass arguments as-is to LLMChunk.ToolCall |
| T-05-03 | Denial of Service | Blocking SDK calls on Dispatchers.Default | mitigate | ALL SDK HTTP calls wrapped in withContext(Dispatchers.IO) — verified by acceptance criteria grep |
| T-05-04 | Denial of Service | Semaphore not per-backend | mitigate | Semaphore is a constructor parameter; each backend instance gets its own Semaphore; Ollama gets Semaphore(Int.MAX_VALUE) by default |
| T-05-05 | Spoofing | LangChain4j Gemini/Ollama type leakage | mitigate | All LangChain4j types (ChatLanguageModel, AiMessage) are fully internal to the adapter; only LLMChunk types cross the boundary |
</threat_model>

// ClaudeBackend.kt
package dev.unityinflow.kore.llm

import com.anthropic.client.AnthropicClient
import com.anthropic.models.MessageCreateParams
import com.anthropic.models.MessageParam
import com.anthropic.models.ContentBlock
import dev.unityinflow.kore.core.ConversationMessage
import dev.unityinflow.kore.core.LLMChunk
import dev.unityinflow.kore.core.LLMConfig
import dev.unityinflow.kore.core.ToolDefinition
import dev.unityinflow.kore.core.port.LLMBackend
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.sync.Semaphore
import kotlinx.coroutines.sync.withPermit
import kotlinx.coroutines.withContext
import kotlinx.serialization.encodeToString
import kotlinx.serialization.json.Json

/**
 * LLM backend adapter for Anthropic Claude using the official Anthropic Java SDK 2.20.0.
 *
 * API keys MUST be provided via constructor — never logged or included in LLMChunk content.
 * All SDK calls are wrapped in [withContext](Dispatchers.IO) to prevent thread pool starvation.
 * [semaphore] is per-backend (not shared with other backends) — per D-18 and Pitfall 11.
 */
class ClaudeBackend(
    private val client: AnthropicClient,
    private val defaultModel: String = "claude-3-5-sonnet-20241022",
    private val semaphore: Semaphore = Semaphore(10),
) : LLMBackend {
    override val name: String = "claude"

    override fun call(
        messages: List<ConversationMessage>,
        tools: List<ToolDefinition>,
        config: LLMConfig,
    ): Flow<LLMChunk> = flow {
        val anthropicMessages = messages
            .filter { it.role != ConversationMessage.Role.System }
            .map { it.toAnthropicMessage() }

        val systemPrompt = messages
            .filter { it.role == ConversationMessage.Role.System }
            .joinToString("\n") { it.content }

        val anthropicTools = tools.map { tool ->
            com.anthropic.models.Tool.builder()
                .name(tool.name)
                .description(tool.description)
                .inputSchema(
                    com.anthropic.models.Tool.InputSchema.builder()
                        .type(com.anthropic.models.Tool.InputSchema.Type.OBJECT)
                        .build()
                )
                .build()
        }

        val response = semaphore.withPermit {
            withContext(Dispatchers.IO) {
                val paramsBuilder = MessageCreateParams.builder()
                    .model(com.anthropic.models.Model.of(config.model.ifBlank { defaultModel }))
                    .maxTokens(config.maxTokens.toLong())
                    .messages(anthropicMessages)

                if (systemPrompt.isNotBlank()) paramsBuilder.system(systemPrompt)
                if (anthropicTools.isNotEmpty()) paramsBuilder.tools(anthropicTools)

                client.messages().create(paramsBuilder.build())
            }
        }

        // Translate provider-specific response to canonical LLMChunk sequence
        response.content().forEach { block ->
            when (block) {
                is ContentBlock.Text -> emit(LLMChunk.Text(block.text()))
                is ContentBlock.ToolUse -> emit(
                    LLMChunk.ToolCall(
                        id = block.id(),
                        name = block.name(),
                        arguments = block.input().toString(),  // JSON object as string
                    )
                )
                else -> { /* ignore unknown content block types */ }
            }
        }

        val usage = response.usage()
        emit(LLMChunk.Usage(
            inputTokens = usage.inputTokens().toInt(),
            outputTokens = usage.outputTokens().toInt(),
        ))
        emit(LLMChunk.Done)
    }

    private fun ConversationMessage.toAnthropicMessage(): MessageParam {
        val roleValue = when (role) {
            ConversationMessage.Role.User, ConversationMessage.Role.Tool -> MessageParam.Role.USER
            ConversationMessage.Role.Assistant -> MessageParam.Role.ASSISTANT
            ConversationMessage.Role.System -> MessageParam.Role.USER  // filtered out above
        }
        return MessageParam.builder()
            .role(roleValue)
            .content(content)
            .build()
    }
}
```

package dev.unityinflow.kore.llm

import kotlinx.coroutines.sync.Semaphore

/**
 * DSL factory functions for LLM backends.
 * Usage: `model = claude(apiKey = System.getenv("ANTHROPIC_API_KEY"))`
 *
 * API keys are passed explicitly — never read from environment directly here.
 * Callers are responsible for retrieving keys from environment variables.
 */

fun claude(
    apiKey: String,
    model: String = "claude-3-5-sonnet-20241022",
    maxConcurrency: Int = 10,
): ClaudeBackend {
    val client = com.anthropic.client.okhttp.AnthropicOkHttpClient.builder()
        .apiKey(apiKey)
        .build()
    return ClaudeBackend(
        client = client,
        defaultModel = model,
        semaphore = Semaphore(maxConcurrency),
    )
}

fun gpt(
    apiKey: String,
    model: String = "gpt-4o",
    maxConcurrency: Int = 10,
): OpenAiBackend {
    val client = com.openai.client.okhttp.OpenAIOkHttpClient.builder()
        .apiKey(apiKey)
        .build()
    return OpenAiBackend(
        client = client,
        defaultModel = model,
        semaphore = Semaphore(maxConcurrency),
    )
}

fun ollama(
    baseUrl: String = "http://localhost:11434",
    model: String = "llama3",
    maxConcurrency: Int = 2,
): OllamaBackend = OllamaBackend(
    baseUrl = baseUrl,
    modelName = model,
    semaphore = Semaphore(maxConcurrency),
)

fun gemini(
    apiKey: String,
    model: String = "gemini-1.5-flash",
    maxConcurrency: Int = 10,
): GeminiBackend = GeminiBackend(
    apiKey = apiKey,
    modelName = model,
    semaphore = Semaphore(maxConcurrency),
)
```
