   ```kotlin
   package dev.unityinflow.kore.observability

   import io.micrometer.core.instrument.Counter
   import io.micrometer.core.instrument.MeterRegistry
   import java.util.concurrent.atomic.AtomicInteger

   /**
    * Kore Micrometer metrics: 4 counters + 1 active-agents gauge (D-21 through D-25).
    *
    * Design rules:
    * - Low cardinality: tag values are configured names (agent_name, model, backend) not UUIDs (D-24)
    * - No per-tool counters: tool-level detail lives in OTel spans only (D-25)
    * - Gauge is a simple AtomicInteger: EventBusMetricsObserver increments/decrements it
    *
    * Counter methods call `.register(registry)` on every call.
    * Micrometer deduplicates by name+tag set — same name+tags always returns the same [Counter].
    */
   class KoreMetrics(private val registry: MeterRegistry) {

       /**
        * Tracks currently active agents. Goes up on [AgentEvent.AgentStarted],
        * down on [AgentEvent.AgentCompleted]. (D-23 — no labels)
        */
       val agentsActive: AtomicInteger = AtomicInteger(0).also { active ->
           registry.gauge("kore.agents.active", active)
       }

       /**
        * Incremented once per completed agent run.
        * Tags: agent_name (configured name), result_type ("success" | "budget_exceeded" | etc.)
        */
       fun agentRunCounter(agentName: String, resultType: String): Counter =
           Counter.builder("kore.agent.runs")
               .tag("agent_name", agentName)
               .tag("result_type", resultType)
               .register(registry)

       /**
        * Incremented once per LLM call.
        * Tags: agent_name, model (e.g. "claude-opus-4-5"), backend (e.g. "claude")
        */
       fun llmCallCounter(agentName: String, model: String, backend: String): Counter =
           Counter.builder("kore.llm.calls")
               .tag("agent_name", agentName)
               .tag("model", model)
               .tag("backend", backend)
               .register(registry)

       /**
        * Incremented by token count per LLM call. Two increments per call: direction=in, direction=out.
        * Tags: agent_name, model, direction ("in" | "out")
        */
       fun tokensUsedCounter(agentName: String, model: String, direction: String): Counter =
           Counter.builder("kore.tokens.used")
               .tag("agent_name", agentName)
               .tag("model", model)
               .tag("direction", direction)
               .register(registry)

       /**
        * Incremented on LLMError and ToolError results.
        * Tags: agent_name, error_type ("llm_error" | "tool_error")
        */
       fun errorCounter(agentName: String, errorType: String): Counter =
           Counter.builder("kore.errors")
               .tag("agent_name", agentName)
               .tag("error_type", errorType)
               .register(registry)
   }
   ```

1. Create `kore-observability/src/main/kotlin/dev/unityinflow/kore/observability/EventBusMetricsObserver.kt`:

   ```kotlin
   package dev.unityinflow.kore.observability

   import dev.unityinflow.kore.core.AgentEvent
   import dev.unityinflow.kore.core.AgentResult
   import dev.unityinflow.kore.core.port.EventBus
   import kotlinx.coroutines.CoroutineScope
   import kotlinx.coroutines.launch
   import java.util.concurrent.ConcurrentHashMap

   /**
    * Subscribes to [EventBus] and drives [KoreMetrics] increments from agent lifecycle events (OBSV-05).
    *
    * agentName lookup: AgentStarted does not carry the agent name in kore-core AgentEvent.
    * EventBusMetricsObserver maintains a [ConcurrentHashMap] populated from AgentStarted.taskId
    * mapped to agent_name from an external name registry or defaulting to "unknown".
    * Full agent_name enrichment will be wired in Phase 3 via Spring context injection.
    *
    * LLM model/backend tags: LLMCallCompleted event does not carry model or backend name.
    * Tags default to "unknown" for now; Phase 3 adds correlation via OTel span context.
    */
   class EventBusMetricsObserver(
       private val eventBus: EventBus,
       private val metrics: KoreMetrics,
       private val scope: CoroutineScope,
       /** Optional: inject agent name resolver. Defaults to returning "unknown". */
       private val agentNameResolver: (agentId: String) -> String = { "unknown" },
   ) {
       // Tracks agentId -> agentName for event correlation (populated on AgentStarted)
       private val agentNames = ConcurrentHashMap<String, String>()

       /**
        * Starts the EventBus subscription in [scope]. Non-blocking.
        * Call once at startup before agents start running.
        */
       fun start() {
           scope.launch {
               eventBus.subscribe().collect { event ->
                   when (event) {
                       is AgentEvent.AgentStarted -> {
                           val name = agentNameResolver(event.agentId)
                           agentNames[event.agentId] = name
                           metrics.agentsActive.incrementAndGet()
                       }
                       is AgentEvent.AgentCompleted -> {
                           val agentName = agentNames.remove(event.agentId) ?: "unknown"
                           metrics.agentsActive.decrementAndGet()
                           metrics.agentRunCounter(agentName, event.result.typeName()).increment()
                           if (event.result is AgentResult.LLMError || event.result is AgentResult.ToolError) {
                               metrics.errorCounter(agentName, event.result.typeName()).increment()
                           }
                       }
                       is AgentEvent.LLMCallCompleted -> {
                           // agentId present; look up name. model/backend unknown until Phase 3.
                           val agentName = agentNames[event.agentId] ?: "unknown"
                           metrics.llmCallCounter(agentName, model = "unknown", backend = "unknown").increment()
                           metrics.tokensUsedCounter(agentName, model = "unknown", direction = "in")
                               .increment(event.tokenUsage.inputTokens.toDouble())
                           metrics.tokensUsedCounter(agentName, model = "unknown", direction = "out")
                               .increment(event.tokenUsage.outputTokens.toDouble())
                       }
                       // ToolCallStarted, ToolCallCompleted, LLMCallStarted: no Micrometer action (D-25)
                       else -> Unit
                   }
               }
           }
       }
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| AgentEvent stream → EventBusMetricsObserver | Internal event bus — all producers are within the JVM process |
| KoreMetrics counters → MeterRegistry exporters | Metrics values exported to external systems (Prometheus, DataDog) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-10 | Denial of Service | agentNames ConcurrentHashMap | mitigate | If AgentCompleted is never emitted (cancelled agent), the map grows unbounded. EventBusMetricsObserver removes on AgentCompleted. Add: also remove on AgentEvent.AgentStarted for duplicate agentIds (idempotent). Add a max-size guard with LRU eviction or remove entries also on any unexpected event type containing an agentId. |
| T-02-11 | Information Disclosure | tag values in metrics (agent_name) | accept | agent_name is a developer-configured string (set in AgentTask.metadata["agent_name"] or DSL). It is intentionally exposed as a tag. If agent names contain PII, that is a misconfiguration concern — document in KDoc. |
| T-02-12 | Denial of Service | EventBus subscription — slow consumer | accept | EventBus uses DROP_OLDEST backpressure (InProcessEventBus design from Phase 1). Observer lag drops oldest metric events rather than backpressuring the agent loop. Acceptable trade-off: a few lost metric increments vs. degraded agent throughput. |
| T-02-13 | Tampering | Micrometer counter values | accept | Counters are in-memory within the JVM process. Tampering requires process-level access. No additional mitigation warranted for v0.1. |
</threat_model>
