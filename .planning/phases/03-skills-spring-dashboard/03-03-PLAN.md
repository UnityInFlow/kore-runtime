4. Create EventBusDashboardObserver.kt (per Pattern 9 from RESEARCH.md):
   ```kotlin
   package dev.unityinflow.kore.dashboard

   import dev.unityinflow.kore.core.AgentEvent
   import dev.unityinflow.kore.core.AgentResult
   import dev.unityinflow.kore.core.port.EventBus
   import kotlinx.coroutines.CoroutineScope
   import kotlinx.coroutines.Dispatchers
   import kotlinx.coroutines.Job
   import kotlinx.coroutines.SupervisorJob
   import kotlinx.coroutines.launch
   import java.time.Instant
   import java.util.concurrent.ConcurrentHashMap

   class EventBusDashboardObserver(private val eventBus: EventBus) {
       private val activeAgents = ConcurrentHashMap<String, AgentState>()
       private val scope = CoroutineScope(SupervisorJob() + Dispatchers.Default)
       // AtomicReference<Job?> as val satisfies CLAUDE.md no-var rule while allowing mutation of the held value
       private val job = java.util.concurrent.atomic.AtomicReference<Job?>(null)

       fun startCollecting() {
           job.set(scope.launch {
               eventBus.subscribe().collect { event ->
                   when (event) {
                       is AgentEvent.AgentStarted -> activeAgents[event.agentId] =
                           AgentState(name = event.agentId, status = AgentStatus.RUNNING, startedAt = Instant.now())
                       is AgentEvent.AgentCompleted -> activeAgents.remove(event.agentId)
                       is AgentEvent.LLMCallCompleted -> activeAgents.compute(event.agentId) { _, state ->
                           state?.copy(tokensUsed = state.tokensUsed + event.tokenUsage.total)
                       }
                       else -> Unit
                   }
               }
           })
       }

       fun snapshot(): Map<String, AgentState> = activeAgents.toMap()

       fun stopCollecting() { job.get()?.cancel() }
   }
   ```
   NOTE: Check AgentEvent.AgentStarted field names against the actual kore-core AgentEvent.kt (read it first). The currentTask field starts empty — AgentStarted only carries agentId and taskId, not the full task content.
   NOTE: TokenUsage.total may need to be `inputTokens + outputTokens` if no `.total` property exists — check TokenUsage.kt before writing this.

   ```kotlin
   ) {

           call.respondHtml {

                   fragmentContainer("active-agents", 2, "$basePath/fragments/active-agents") {}
                   fragmentContainer("recent-runs", 5, "$basePath/fragments/recent-runs") {}
                   fragmentContainer("cost-summary", 10, "$basePath/fragments/cost-summary") {}
               }
           }
       }

       get("$basePath/fragments/active-agents") {
           call.respondHtml { activeAgentsFragment(observer.snapshot()) }
       }

       get("$basePath/fragments/recent-runs") {
           // Ktor 3.2 CIO route handlers are suspend — call suspend functions directly, no runBlocking
           val runs = dataService.getRecentRuns(20)
           call.respondHtml { recentRunsFragment(runs, degraded = !dataService.hasStorage()) }
       }

       get("$basePath/fragments/cost-summary") {
           val costs = dataService.getCostSummary()
           call.respondHtml { costSummaryFragment(costs, degraded = !dataService.hasStorage()) }
       }
   }
   ```

   ```kotlin
   package dev.unityinflow.kore.dashboard

   import dev.unityinflow.kore.core.port.AuditLog
   import dev.unityinflow.kore.core.port.EventBus
   import io.ktor.server.cio.*
   import io.ktor.server.engine.*
   import org.springframework.context.SmartLifecycle

   class DashboardServer(
       private val eventBus: EventBus,
       private val auditLog: AuditLog?,
       private val config: DashboardConfig = DashboardConfig(),
   ) : SmartLifecycle {
       // AtomicReference<EmbeddedServer<*, *>?> satisfies CLAUDE.md no-var rule
       private val engine = java.util.concurrent.atomic.AtomicReference<EmbeddedServer<*, *>?>(null)
       private val observer = EventBusDashboardObserver(eventBus)
       private val dataService = DashboardDataService(auditLog)

       override fun start() {
           observer.startCollecting()
           engine.set(embeddedServer(CIO, port = config.port) {
               routing {
                   configureDashboardRoutes(observer, dataService, config)
               }
           }.also { it.start(wait = false) })
       }

       override fun stop() {
           engine.get()?.stop(gracePeriodMillis = 1000, timeoutMillis = 5000)
           observer.stopCollecting()
           engine.set(null)
       }

       override fun isRunning(): Boolean = engine.get() != null
       override fun isAutoStartup(): Boolean = config.enabled
       override fun getPhase(): Int = Int.MAX_VALUE - 1
   }
   ```

For this plan, HTMX attributes are added manually via kotlinx.html attributes() — no need to install the HTMX plugin unless it provides required DSL features.
